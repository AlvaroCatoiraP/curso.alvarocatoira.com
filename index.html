Capítulo 15 · POO

Ejercicio 12

Pendiente

Crea una clase Persona con un atributo nombre y un método saludar.

class Persona:
    def __init__(self, nombre):
        self.nombre = nombre

    def saludar(self):
        print(f"Hola, soy {self.nombre}")

p = Persona("Alvaro")
p.saludar()
